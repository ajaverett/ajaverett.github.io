%%manim -qm Scene3

class Scene3(Scene):
    def construct(self):
        plane = NumberPlane()
        self.add(plane)

        red_dot = Dot(color=RED)
        green_dot = Dot(color=GREEN)
        green_dot.next_to(red_dot, RIGHT + UP)
        self.add(red_dot, green_dot)

        s = Square(color=ORANGE)
        s.shift(UP*2, RIGHT*4)
        self.add(s)

        c = Circle(color=PURPLE)
        c.move_to([-3,-2,0])

        c2 = Circle(radius=.5, color=RED, fill_opacity=.5)
        c3 = c2.copy().set_color(YELLOW)
        c4 = c2.copy().set_color(ORANGE)

        c2.align_to(s, UP)
        c3.align_to(s, RIGHT)
        c4.align_to(s, UP + RIGHT)
        self.add(c3)

        self.play(ReplacementTransform(c, c2))
        self.play(ReplacementTransform(c3, c4))

        self.wait(3)
